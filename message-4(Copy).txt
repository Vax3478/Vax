-- OfflinePlayerLookup only (no loadOfflinePlayer / no gifting UI / no camera)
function getPlayerGamepasses(username)
	local userId
	local okId, idErr = pcall(function()
		userId = Players:GetUserIdFromNameAsync(username)
	end)
	if not okId then
		return nil, "invalid username: " .. tostring(idErr)
	end
	local booth
	local okLookup, lookupErr = pcall(function()
		booth = Remotes.Function("OfflinePlayerLookup"):InvokeServer(userId)
	end)
	if not okLookup then
		return nil, "OfflinePlayerLookup failed: " .. tostring(lookupErr)
	end
	if typeof(booth) ~= "Instance" then
		return nil, "unexpected lookup result: " .. typeof(booth)
	end
	local boothUI = booth:FindFirstChild("BoothUI", true)
	local itemsFrame = boothUI and boothUI:FindFirstChild("Items") and boothUI.Items:FindFirstChild("Frame")
	local gamepasses = {}
	if itemsFrame then
		for _, child in ipairs(itemsFrame:GetChildren()) do
			if child:GetAttribute("AssetType") == "Gamepass" then
				local prompt = child:FindFirstChild("Prompt")
				table.insert(gamepasses, {
					name = child.Name,
					assetId = child:GetAttribute("AssetId"),
					assetPrice = child:GetAttribute("AssetPrice"),
					assetType = child:GetAttribute("AssetType"),
					prompt = prompt,
					item = child,
				})
			end
		end
	end
	table.sort(gamepasses, function(a, b)
		return (a.assetPrice or 0) < (b.assetPrice or 0)
	end)
	local offlineBooths = ReplicatedStorage:FindFirstChild("OfflineBooths")
	if offlineBooths then
		booth.Parent = offlineBooths
	end
	return {
		username = username,
		userId = userId,
		boothName = booth.Name,
		booth = booth,
		gamepasses = gamepasses,
	}
end
